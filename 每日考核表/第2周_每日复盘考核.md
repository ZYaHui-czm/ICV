```python
# 字典 = 键值对集合

# ── 查 ──
print(d["name"])              # "小明"
print(d.get("score"))         # None（键不存在不报错）
print(d.get("score", 0))      # 0（不存在时返回默认值）

# ── 增/改 ──
d["score"] = 90               # 键不存在→新增；键存在→修改

# ── 删 ──
d.pop("age")                  # 删除键"age"并返回其值
del d["name"]                 # 删除键（不返回值）

# ── 判断键是否存在 ──
if "name" in d:               # ✅ 推荐
    print("有name这个键")

# 遍历所有键

    print(key)

# 遍历所有值
for value in d.values():
    print(value)

# 遍历键值对（最常用！）
for key, value in d.items():
    print(f"{key} → {value}")

# get() vs 方括号[]
d = {"name": "小明"}
print(d["score"])             # ❌ KeyError！
print(d.get("score"))         # None
print(d.get("score", 0))      # 0

# 键、值、键值对
d.keys()                      # dict_keys(['a', 'b', 'c'])
d.values()                    # dict_values([1, 2, 3])
d.items()                     # dict_items([('a',1), ('b',2), ('c',3)])

# 合并字典
d.update({"d": 4, "e": 5})

# 清空
d.clear()

# 长度
len(d)                        # 键值对个数

# 实战：词频统计
text = "apple banana apple orange banana apple"

count = {}

    count[word] = count.get(word, 0) + 1

```

```python
# 元组（tuple）— 不可变的列表
t = (1, 2, 3)
t2 = 4, 5, 6              # 括号可以省略

point = (3, 5)

a, b = 1, 2
a, b = b, a               # a=2, b=1

# 函数返回多个值（本质是返回元组）
def get_info():
    return "小明", 18, 85
name, age, score = get_info()

# enumerate 背后就是元组拆包
for i, val in enumerate(["a", "b"]):

s = {1, 2, 3}

s.remove(2)                # {1, 3, 4}
print(3 in s)              # True

# ⚠️ 集合无序，不能索引
# s[0]   # TypeError!

# 集合运算 — 最强大的功能
a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

a | b   # 并集：{1, 2, 3, 4, 5, 6}
a & b   # 交集：{3, 4}
a - b   # 差集：{1, 2}  （a有b没有的）
a ^ b   # 对称差：{1, 2, 5, 6}  （各自独有的）

# 列表去重一行搞定
nums = [3, 1, 4, 1, 5, 9, 2, 6, 5, 3]
```

```python
matrix = [

students = [

# 字典套字典（分组信息）
school = {
    "一班": {"人数": 30, "班主任": "张老师"},
    "二班": {"人数": 28, "班主任": "李老师"}
}
print(school["一班"]["班主任"])  # "张老师"

# 字典套列表（分类数据）
grades = {
    "小明": [85, 90, 78],
    "小红": [92, 88, 95]
}
print(grades["小明"][1])   # 90    小明的第二科成绩

# 遍历嵌套结构
for student in students:
    print(f"{student['name']}：{student['score']}分")

for name, scores in grades.items():
    avg = sum(scores) / len(scores)
    print(f"{name} 平均分：{avg:.1f}")

# 实战：学生成绩管理系统
def add_student(db, name, score):
    if name not in db:
        db[name] = []
    db[name].append(score)

def get_avg(db, name):
    scores = db.get(name, [])
    if not scores:
        return None
    return sum(scores) / len(scores)

db = {}
add_student(db, "小明", 85)
add_student(db, "小明", 90)
add_student(db, "小红", 92)
print(db)                    # {'小明': [85, 90], '小红': [92]}
print(get_avg(db, "小明"))   # 87.5
print(get_avg(db, "小刚"))   # None
```

**口诀**：
> 列表套字典是表格，一行一个字典存
> 字典套列表是分类，一个键对应一列值
> 访问先外层再内层，[键][索引]顺序别颠倒
> 修改嵌套先判断，key不存在别硬来

**常见坑**：
| 错误 | 正确 | 原因 |
|------|------|------|
| `db[key]` 不存在就写 | 先判断 `if key not in db` 或 `db.setdefault()` | 不存在的key会KeyError |
| `students["小明"][0]` | `students[0]["score"]` | 字典用键访问，列表用索引 |
| 修改嵌套列表影响原数据 | 用深拷贝 `copy.deepcopy()` | 嵌套结构浅拷贝只复制外层 |

```python
def class_report(date):
    date_report = {}
    for course, scores in date.items():
        date_report[course] = {
            "avg": round(sum(scores) / len(scores), 1),
            "max": max(scores)
        }
    return date_report
```

## 🗓️ 第11天 <a id="day11"></a>

**主题**：文件 I/O

**学习内容**：
```python
# 方式1：open + close（手动关，容易忘）
f = open("data.txt", "r", encoding="utf-8")
content = f.read()
f.close()               # ❌ 容易忘记

# 方式2：with open（自动关，✅ 推荐）
with open("data.txt", "r", encoding="utf-8") as f:
    content = f.read()
# 缩进结束，文件自动关闭

# ① 一次读完整个文件
with open("data.txt", "r", encoding="utf-8") as f:
    content = f.read()          # 整个文件 → 一个字符串
    print(content)

# ② 按行读成列表
with open("data.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()       # 每一行 → 列表的一个元素
    print(lines)                # ["第一行\n", "第二行\n", ...]

# ③ 逐行读取（大文件推荐，不占内存）
with open("data.txt", "r", encoding="utf-8") as f:
    for line in f:              # f 本身就是可迭代的！
        print(line.strip())     # strip() 去掉换行符 \n

# "w" → 覆盖写（会清空原文件！）
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("Hello\n")
    f.write("World\n")
    f.writelines(["a\n", "b\n", "c\n"])

# "a" → 追加写（在末尾添加，不清空）
with open("output.txt", "a", encoding="utf-8") as f:
    f.write("这行会追加到末尾\n")

# 实战：文件版笔记
def save_note(filename, note):
    with open(filename, "a", encoding="utf-8") as f:
        f.write(note + "\n")
    print("笔记已保存")

def read_notes(filename):
    try:
        with open(filename, "r", encoding="utf-8") as f:
            notes = f.readlines()
            if not notes:
                print("暂无笔记")
            else:
                for i, note in enumerate(notes, 1):
                    print(f"{i}. {note.strip()}")
    except FileNotFoundError:
        print("文件不存在，暂无笔记")

# 处理文件不存在的异常
import os
if os.path.exists("不存在.txt"):
    with open("不存在.txt", "r", encoding="utf-8") as f:
        print(f.read())
else:
    print("文件不存在")

# 或者用 try（更推荐）
try:
    with open("不存在.txt", "r", encoding="utf-8") as f:
        print(f.read())
except FileNotFoundError:
    print("文件不存在，请先创建")
```

**文件模式表格**：
| 模式 | 作用 | 文件不存在？ | 文件已存在？ |
|:--:|------|:--:|:--:|
| `"r"` | 读 | ❌ 报错 | ✅ |
| `"w"` | 写（覆盖） | ✅ 创建 | ❌ 清空原内容 |
| `"a"` | 写（追加） | ✅ 创建 | ✅ 在末尾追加 |
| `"r+"` | 读+写 | ❌ 报错 | ✅ |

**常见错误表格**：
| 错误 | 正确 | 原因 |
|------|------|------|
| 忘记 `encoding="utf-8"` | 始终加上 | 中文会乱码 |
| 用 `"w"` 模式读文件 | 用 `"r"` 读 | w 会清空文件内容 |
| `readlines()` 忘记 `.strip()` | `line.strip()` 去掉 `\n` | 每行末尾有换行符 |
| 文件路径用 `\` | 用 `/` 或 `\\` 或原始字符串 `r"path"` | `\n` 会被当作换行转义 |

**口诀**：
> with open 最安全，自动关闭不用管
> r 读 w 写 a 追加，encoding 加 utf-8

**主题**：异常处理

**学习内容**：
```python
# 没有异常处理 — 一个错误，全盘崩溃
num = int(input("输入数字："))
result = 100 / num
print(f"结果是{result}")
# 如果用户输入 0 或 "abc" → 程序直接挂掉

# try-except — 基本结构
try:
    num = int(input("输入数字："))
    result = 100 / num
    print(f"结果是{result}")
except ValueError:
    print("请输入一个数字！")
except ZeroDivisionError:
    print("不能除以0！")

# 一个except捕获多个异常
try:
    x = int(input())
    print(100 / x)
except (ValueError, ZeroDivisionError) as e:
    print(f"出错了：{e}")

# else — 没出异常才执行
try:
    num = int(input("输入一个数字："))
    result = 100 / num
except ValueError:
    print("不是数字")
except ZeroDivisionError:
    print("不能除0")
else:
    print(f"结果是 {result}")

# finally — 无论是否异常都会执行
f = None
try:
    f = open("data.txt", "r", encoding="utf-8")
    content = f.read()
    print(content)
except FileNotFoundError:
    print("文件不存在")
finally:
    if f:
        f.close()
    print("执行完毕")

# raise — 手动抛出异常
def set_age(age):
    if age < 0 or age > 150:
        raise ValueError("年龄必须在0-150之间")
    print(f"年龄设置为 {age}")
```

**常见异常速查表**：
| 异常类型 | 何时触发 |
|---------|---------|
| `ValueError` | `int("abc")` — 类型转换失败 |
| `ZeroDivisionError` | `1 / 0` |
| `FileNotFoundError` | `open("不存在.txt")` |
| `KeyError` | `d["不存在的键"]` |
| `IndexError` | `lst[999]` 索引超范围 |
| `TypeError` | `"a" + 1` 类型不配 |
| `SyntaxError` | 代码语法错误（不能 try 捕获） |

**口诀**：
> try 里面写代码，except 捕获错
> 具体异常先捕获，笼统异常放最后
> else 没出错才执行，finally 死活都执行
> raise 手动抛异常，可以自定义错误检查
