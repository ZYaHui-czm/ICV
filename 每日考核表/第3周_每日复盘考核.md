### 四、手写代码（30分）
**题目**：save_diary(entry) — JSON格式追加日记，不存在则创建空列表

```python
def save_diary(entry):
    now = datetime.now().strftime("%Y-%m-%d %H:%M")
    diary = {"time": now, "content": entry}
    with open("diary.json", "a", encoding="utf-8") as dia:
        json.dump(diary, dia, ensure_ascii=False, indent=2)
```
🔴 用 `"a"` + `json.dump` 会产生 `{...}{...}{...}` 不合法JSON
🔴 题目要求"不存在创建空列表"，这里没处理

✅ 正确写法：
```python
def save_diary(entry):
    import os
    # 读取已有数据（不存在则空列表）
    if os.path.exists("diary.json"):
        with open("diary.json", "r", encoding="utf-8") as f:
            diary_list = json.load(f)
    else:
        diary_list = []
    # 追加新条目
    diary_list.append({
        "time": datetime.now().strftime("%Y-%m-%d %H:%M"),
        "content": entry
    })
    # 完整写回
    with open("diary.json", "w", encoding="utf-8") as f:
        json.dump(diary_list, f, ensure_ascii=False, indent=2)
```

**学习内容**：
```python
# lambda — 无名函数，一行搞定
add = lambda x, y: x + y
square = lambda n: n ** 2

# lambda 作为 sorted/max 的 key
students = [("小明", 85), ("小红", 92), ("小刚", 78)]
sorted(students, key=lambda s: s[1])             # 按成绩升序
sorted(students, key=lambda s: s[1], reverse=True)  # 按成绩降序
names = ["apple", "banana", "kiwi", "pear"]
max(names, key=lambda n: len(n))                # "banana"（最长的）

# map() — 对每个元素做同样的事
nums = [1, 2, 3, 4]
squares = list(map(lambda x: x ** 2, nums))     # [1, 4, 9, 16]
words = ["hello", "world"]
upper_words = list(map(str.upper, words))       # ['HELLO', 'WORLD']
str_nums = ["1", "2", "3", "4"]
int_nums = list(map(int, str_nums))             # [1, 2, 3, 4] 批量转数字

# filter() — 挑选符合条件的
nums = [1, 2, 3, 4, 5, 6]
evens = list(filter(lambda x: x % 2 == 0, nums))  # [2, 4, 6]
words = ["cat", "elephant", "dog", "hippopotamus"]
long_words = list(filter(lambda w: len(w) > 3, words))  # ['elephant', 'hippopotamus']
items = [0, 1, "", "hello", None, [], [1, 2]]
truthy = list(filter(None, items))              # [1, 'hello', [1, 2]]

# sorted() — 灵活排序
nums = [3, 1, 4, 1, 5, 9]
sorted(nums)                  # [1, 1, 3, 4, 5, 9]
sorted(nums, reverse=True)    # [9, 5, 4, 3, 1, 1]
words = ["apple", "banana", "kiwi", "pear"]
sorted(words, key=len)        # 按长度 → ['kiwi', 'pear', 'apple', 'banana']
pairs = [("a", 3), ("b", 1), ("c", 2)]
sorted(pairs, key=lambda x: x[1])  # [('b', 1), ('c', 2), ('a', 3)]
d = {"apple": 3, "banana": 1, "kiwi": 2}
sorted(d.items(), key=lambda x: x[1])  # 字典按值排序
```

**三者对比表**：
| | 作用 | 返回 |
|--|------|------|
| `map(f, lst)` | 每个元素变一下 | 相同数量 |
| `filter(f, lst)` | 只留符合条件的 | 可能减少 |
| `sorted(lst, key=f)` | 重新排列 | 相同数量 |

**学习内容**：
```python
# 定义类 — 创建对象的"蓝图"
class Student:
    school = "第一中学"        # 类变量：所有对象共享

    def __init__(self, name, age, score):
        """初始化方法：创建对象时自动调用"""
        self.name = name       # 实例变量：每个对象独有
        self.age = age
        self.score = score

    def introduce(self):
        """普通方法：对象的行为"""
        print(f"我叫{self.name}，今年{self.age}岁，成绩{self.score}分")

    def is_pass(self):
        """判断是否及格"""
        return self.score >= 60

# 创建对象（实例化）— __init__ 自动触发
s1 = Student("小明", 18, 85)
s2 = Student("小红", 17, 92)

s1.introduce()    # 我叫小明，今年18岁，成绩85分
print(s2.is_pass())  # True

# 访问和修改属性
print(s1.name)    # "小明"
s1.score = 95     # 修改属性

# self 到底是谁？哪个对象调用，self 就指向哪个
class Dog:
    def __init__(self, name):
        self.name = name
    def bark(self):
        print(f"{self.name} 汪汪！")

d1 = Dog("旺财")
d2 = Dog("小黑")
d1.bark()  # self → d1 → "旺财 汪汪！"
d2.bark()  # self → d2 → "小黑 汪汪！"
# d1.bark() 等价于 Dog.bark(d1)

# __init__ — 创建对象时自动执行
class Book:
    def __init__(self, title, author, pages=0):
        self.title = title
        self.author = author
        self.pages = pages
        print(f"《{self.title}》已创建")

b = Book("Python入门", "张三", 300)  # ▶《Python入门》已创建

# 类变量共享
Student.school = "第二中学"  # 改类变量
print(s1.school)  # "第二中学"（全部跟着变）
print(s2.school)  # "第二中学"
```

**类变量 vs 实例变量**：
| | 类变量 | 实例变量 |
|--|--------|---------|
| 定义位置 | `class`下直接写 | `__init__`里`self.xxx` |
| 共享？ | ✅ 所有对象共用 | ❌ 每人独立 |
| 修改 | `类名.变量=` | `self.变量=` |

**常见错误**：
| ❌ 错误 | ✅ 正确 | 原因 |
|--------|--------|------|
| 方法忘写 `self` | `def f(self):` | 第一个参数必须是self |
| `__init__` 写成 `__init` | `__init__` | 双下划线前后各两个 |
| 方法里用 `name` 不用 `self.name` | `self.name` | 不加self是局部变量 |
| 调用方法忘加括号 | `s1.introduce()` | 必须 `()` |

**口诀**：
> class 定义蓝图，init 用来造对象
> self 指自己，哪个调用就指哪个
> 方法必带 self，属性挂在 self 上
> 类变量大家共享，实例变量每人一份

## 🗓️ 第18天 <a id="day18"></a>

**主题**：特殊方法 + 继承

**学习内容**：
```python
# 特殊方法 — 让对象"会说话"
class Student:
    def __init__(self, name, score):
        self.name = name
        self.score = score

    def __str__(self):
        """print(对象) 和 str(对象) 时调用"""
        return f"学生：{self.name}，成绩：{self.score}"

    def __repr__(self):
        """在列表里、调试时显示"""
        return f"Student('{self.name}', {self.score})"

    def __eq__(self, other):
        """== 比较时调用"""
        return self.score == other.score

    def __lt__(self, other):
        """< 比较时调用（排序需要）"""
        return self.score < other.score

s1 = Student("小明", 85)
s2 = Student("小红", 92)
print(s1)           # 学生：小明，成绩：85（__str__）
print(s1 == s2)     # False（__eq__）
print(s1 < s2)      # True（__lt__）

# 常用特殊方法
# __str__ → print(obj)   __repr__ → 调试/列表   __eq__ → ==
# __lt__ → <排序   __len__ → len(obj)   __getitem__ → obj[0]
# __contains__ → x in obj

# 继承 — "儿子继承爸爸"
class Person:                       # 父类（基类）
    def __init__(self, name, age):
        self.name = name
        self.age = age
    def introduce(self):
        print(f"我叫{self.name}，今年{self.age}岁")

class Student(Person):              # 子类继承 Person
    def __init__(self, name, age, score):
        super().__init__(name, age)  # 调用父类 __init__
        self.score = score
    def introduce(self):             # 重写父类方法
        super().introduce()          # 还能调父类原版
        print(f"成绩：{self.score}")

class Teacher(Person):
    def __init__(self, name, age, subject):
        super().__init__(name, age)
        self.subject = subject

# super() — 调用父类
class Animal:
    def __init__(self, name):
        self.name = name

class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name)       # 父类设置 self.name
        self.breed = breed           # 子类自己的

d = Dog("旺财", "金毛")
print(d.name)   # "旺财"

# isinstance / issubclass
isinstance(s, Student)   # True
isinstance(s, Person)    # True（子类实例也是父类实例）
issubclass(Student, Person)  # True
```

**口诀**：
> __str__给人看，__repr__调试用；__eq__判相等，__lt__比大小
> 继承父子类，super叫爸爸；子类可重写，父类方法还能调

```python
import time
from functools import wraps

# 装饰器就是一个函数：接收函数，返回新函数
def timer(func):
    def wrapper():
        start = time.time()
        func()                    # 调用原函数
        end = time.time()
        print(f"耗时：{end - start:.2f}秒")
    return wrapper                # 返回包装后的函数

@timer   # 等价于 slow_function = timer(slow_function)
def slow_function():
    time.sleep(0.5)
    print("完成")

slow_function()  # 完成 \n 耗时：0.50秒

# 处理带参数的函数 — *args, **kwargs 接收任意参数
def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)  # 透传给原函数，保留返回值
        end = time.time()
        print(f"耗时：{end - start:.2f}秒")
        return result
    return wrapper

@timer
def add(a, b):
    time.sleep(0.2)
    return a + b

print(add(3, 5))  # 耗时：0.20秒 \n 8

# 带参数的装饰器 — 多套一层
def timer_with_unit(unit="秒"):
    def decorator(func):
        def wrapper(*args, **kwargs):
            start = time.time()
            result = func(*args, **kwargs)
            end = time.time()
            elapsed = end - start
            if unit == "毫秒":
                elapsed *= 1000
            print(f"耗时：{elapsed:.2f}{unit}")
            return result
        return wrapper
    return decorator

@timer_with_unit(unit="毫秒")
def quick():
    time.sleep(0.05)

quick()  # 耗时：50.00毫秒

# 保留原函数信息 — @wraps
def timer(func):
    @wraps(func)          # 保留 __name__ 和 __doc__
    def wrapper(*args, **kwargs):
        ...
    return wrapper

@timer
def my_func():
    """这是文档"""
    pass

print(my_func.__name__)  # "my_func"（不加wraps会变成"wrapper"）

# 常见装饰器
# 日志装饰器
def log(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}({args}, {kwargs})")
        result = func(*args, **kwargs)
        print(f"返回 {result}")
        return result
    return wrapper

# 重试装饰器
def retry(max_times=3):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_times + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    print(f"第{attempt}次失败：{e}")
                    if attempt == max_times:
                        raise
        return wrapper
    return decorator
```

**口诀**：
> 装饰器就是包装函数，不碰原函数加功能
> @符号糖衣，func = decorator(func)
> *args **kwargs 透传参数，return result 送回返回值
> functools.wraps 保名字，带参装饰器多套一层

**得分**：100/100 🟢🏆 满分！

```python
from functools import wraps

def cache(func):
    memory = {}
    @wraps(func)
    def wrapper(*args):
        if args in memory:          # 查缓存
            return memory[args]     # 命中 → 直接返回
        result = func(*args)        # 未命中 → 计算
        memory[args] = result       # 存入缓存
        return result
    return wrapper
```
